// 쿠키 삭제 - 만기 시간을 과거로 되돌린다. function delCookie() { var date = new Date(); date.setDate(date.getDate() - 1); var setCookie = ''; setCookie += 'CookieName=ABC;'; setCookie += 'Expires=' + date.toUTCString(); document.cookie = setCookie; } document.getElementById('delCookie').addEventListener('click', function () { delCookie(); alert("Deleted"); });
